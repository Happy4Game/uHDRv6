462    def setValue(self, key, value, autoScale=False):
463        if pref.verbose: print(" [MODEL] >> ToneCourveModel.setValue(",key,", ",value,", autoScale=",autoScale,")")
464        value = int(value)
465        # check key
466        if key in self.control.keys():
467            # transform in list
468            listKeys = list(self.control.keys())
469            listValues = np.asarray(list(self.control.values()))
470            index = listKeys.index(key)
471
472            if (listValues[:index,1] <= value).all() and (value <= listValues[index+1:,1]).all():
473                # can change
474                oldValue = self.control[listKeys[index]]
475                self.control[listKeys[index]] = [oldValue[0],value]            
476            elif not (value <= listValues[index+1:,1]).all():
477                if autoScale:
478                    minValue = min(listValues[index:,1])
479                    maxValue = listValues[-1:,1]
480                    u = (listValues[index+1:,1] -minValue)/(maxValue-minValue)
481
482                    newValues = value*(1- u)+ u*maxValue
483                    for i,v in enumerate(newValues):
484                        oldValue = self.control[listKeys[i+index+1]]
485                        self.control[listKeys[i+index+1]] = [oldValue[0],np.round(v)]
486                else:
487                    # not autoScale, set to minValue
488                    oldValue = self.control[listKeys[index]]
489                    minValue = min(listValues[index+1:,1])
490                    self.control[listKeys[index]] = [oldValue[0],minValue]
491            elif not (listValues[:index,1] <= value).all():
492                if autoScale:
493                    minValue = listValues[0,1]
494                    maxValue = max(listValues[:index,1])
495                    u = (listValues[:index,1] -minValue)/(maxValue-minValue)
496                    newValues = minValue*(1- u)+ u*value
497                    for i,v in enumerate(newValues):
498                        oldValue = self.control[listKeys[i]]
499                        self.control[listKeys[i]] = [oldValue[0],np.round(v)]
500                else:
501                    # not autoScale, set to maxValue
502                    oldValue = self.control[listKeys[index]]
503                    maxValue =  max(listValues[:index,1])
504                    self.control[listKeys[index]] = [oldValue[0],maxValue]
505
506        return self.control
