 40        self.label.resize(self.size())
 41        self.label.setPixmap(self.imagePixmap.scaled(self.size(),Qt.KeepAspectRatio))

 44        self.resize()

 47    def setPixmap(self,colorData):
 48        if not isinstance(colorData, np.ndarray): 
 49            colorData = ImageWidgetView.emptyImageColorData()
 50        # self.colorData = colorData
 51        height, width, channel = colorData.shape   # compute pixmap
 52        bytesPerLine = channel * width
 53        # clip
 54        colorData[colorData>1.0] = 1.0
 55        colorData[colorData<0.0] = 0.0
 56        ##
 57        #t_begin = datetime.now()
 58        ##
 59        qImg = QImage((colorData*255).astype(np.uint8), width, height, bytesPerLine, QImage.Format_RGB888) # QImage
 60        self.imagePixmap = QPixmap.fromImage(qImg)
 61        self.resize()
 62        ###
 63        ##t_end = datetime.now()
 64        ##print(" >> ImageWidgetView.setPixmap()","[",t_end-t_begin,"ms]")
 65        ###
 66        return self.imagePixmap
 67
 68    def setQPixmap(self, qPixmap):
 69        self.imagePixmap = qPixmap
 70        self.resize()

160    def changePageNumber(self,step):
161        print(" [VIEW] >> ImageGalleryView.changePageNumber(",step,")")
162
163        nbImagePerPage = controller.GalleryMode.nbRow(self.shapeMode)*controller.GalleryMode.nbCol(self.shapeMode)
164        maxPage = ((len(self.controller.model.processPipes)-1)//nbImagePerPage) + 1
165
166        oldPageNumber = self.pageNumber
167        if (self.pageNumber+step) > maxPage-1:
168            self.pageNumber = 0
169        elif (self.pageNumber+step) <0:
170            self.pageNumber = maxPage-1
171        else:
172            self.pageNumber  = self.pageNumber+step
173        self.controller.model.loadPage(self.pageNumber)
174        self.updateImages()
175        print(" [VIEW] >> ImageGalleryView.changePageNumber(currentPage:",self.pageNumber," | max page:",maxPage,")")
176
177    def updateImages(self):
178        print(" [VIEW] >> ImageGalleryView.updateImages(",")")
179
180        """ update images content """
181        nbImagePerPage = controller.GalleryMode.nbRow(self.shapeMode)*controller.GalleryMode.nbCol(self.shapeMode)
182        maxPage = ((len(self.controller.model.processPipes)-1)//nbImagePerPage) + 1
183
184        index=0
185        for i in range(controller.GalleryMode.nbRow(self.shapeMode)): 
186            for j in range(controller.GalleryMode.nbCol(self.shapeMode)):
187                # get image controllers                                                                                         
188                iwc = self.imagesControllers[index]
189                idxImage = index+nbImagePerPage*self.pageNumber # image index
190                if (idxImage) <len(self.controller.model.processPipes):                                   
191                    iwc.setImage(self.controller.model.processPipes[idxImage].getImage())
192                else: iwc.view.setPixmap(ImageWidgetView.emptyImageColorData())                                                 
193                index +=1                                                                                                       
194        self.pageNumberLabel.setText(str(self.pageNumber)+"/"+str(maxPage-1))

207        imageIndex = 0
208        for i in range(controller.GalleryMode.nbRow(self.shapeMode)): 
209            for j in range(controller.GalleryMode.nbCol(self.shapeMode)):
210                iwc = controller.ImageWidgetController(id=imageIndex)
211                self.imagesControllers.append(iwc)
212                self.imagesLayout.addWidget(iwc.view,i,j)
213                imageIndex +=1

215    def wheelEvent(self, event):
216        print(" [EVENT] >> ImageGalleryView.wheelEvent(",")")
217
218        if event.angleDelta().y() < 0 :     
219            self.changePageNumber(+1)
220            if self.shapeMode == controller.GalleryMode._1x1 : 
221                self.controller.selectImage(0)
222
223        else :                              
224            self.changePageNumber(-1)
225            if self.shapeMode == controller.GalleryMode._1x1 : 
226                self.controller.selectImage(0)
227        event.accept()
